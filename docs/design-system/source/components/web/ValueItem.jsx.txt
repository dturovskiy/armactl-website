import React from 'react';

const VI_CSS = `
.ac-value-item { font-family: var(--font-mono); position: relative; padding-left: 4.5rem; }
.ac-value-item__num { position: absolute; left: 0; top: -0.4rem; font-size: var(--text-3xl);
  font-weight: var(--weight-semibold); color: var(--neon-orange); opacity: 0.18; line-height: 1; }
.ac-value-item__title { font-size: var(--text-md); font-weight: var(--weight-semibold);
  letter-spacing: var(--tracking-wide); text-transform: uppercase;
  color: var(--neon-orange); margin: 0 0 0.5rem 0; }
.ac-value-item__body { margin: 0; font-size: var(--text-sm); font-weight: var(--weight-light);
  color: var(--text-main); line-height: var(--leading-base); }
`;

function injectValueItemCss() {
  if (typeof document === 'undefined' || document.getElementById('ac-value-item-css')) return;
  const el = document.createElement('style');
  el.id = 'ac-value-item-css';
  el.textContent = VI_CSS;
  document.head.appendChild(el);
}

/**
 * "Operational Philosophy" value item — ghost number, orange uppercase title.
 */
export function ValueItem({ number, title, children, style }) {
  injectValueItemCss();
  return (
    <div className="ac-value-item" style={style}>
      <span className="ac-value-item__num">{number}</span>
      <h4 className="ac-value-item__title">{title}</h4>
      <p className="ac-value-item__body">{children}</p>
    </div>
  );
}
